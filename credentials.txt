#!/bin/bash

echo "--- 1. Installing Docker ---"
# Update the package list
sudo apt update -y

# Install packages to allow apt to use a repository over HTTPS
sudo apt install -y apt-transport-https ca-certificates curl software-properties-common

# Add Docker's official GPG key
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

# Add the Docker repository to APT sources
echo "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

# Update package list again (to include Docker repo)
sudo apt update -y

# Install the latest version of Docker Engine and Docker Compose
sudo apt install -y docker-ce docker-ce-cli containerd.io

# --- CRITICAL: Add the current user (ubuntu) to the docker group ---
RUNNER_USER=$(whoami)
echo "Adding user '$RUNNER_USER' to the 'docker' group."
sudo usermod -aG docker $RUNNER_USER
echo "Docker installation complete. You MUST log out and log back in for group changes to take effect!"

echo "--- 2. Installing AWS CLI v2 ---"
# Install unzip, required for AWS CLI installation
sudo apt install -y unzip

# Download AWS CLI bundle
curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o "awscliv2.zip"

# Unzip and install
unzip awscliv2.zip
sudo ./aws/install --update

# Clean up
rm -rf awscliv2.zip aws/

echo "AWS CLI installation complete."

echo "--- 3. Verifying Installation ---"
# Check if commands are available (Note: You may still need to log out/in for 'docker' to work)
docker --version
aws --version
echo "--- Setup Script Finished ---"
echo "--- IMPORTANT NEXT STEP: Log out of this SSH session and then log back in to apply Docker permissions. ---"
